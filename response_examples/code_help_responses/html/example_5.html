Pandas Styler - Highlight Values Above Average

This code demonstrates how to create a function using the pandas styler to color the values above the average for a column in a dataframe.

Code


import pandas as pd

def highlight_above_average(df):
    """
    Function to highlight values above the average for each column in a dataframe.
    """
    # Calculate the average for each column
    avg = df.mean()

    # Create a styler object for the dataframe
    styler = df.style

    # Iterate over each column
    for col in df.columns:
        # Highlight values above the average for the column
        styler = styler.apply(lambda x: ['background-color: yellow' if val > avg[col] else '' for val in x], subset=col)

    return styler

# Example usage
data = {'A': [1, 2, 3, 4, 5],
        'B': [6, 7, 8, 9, 10],
        'C': [11, 12, 13, 14, 15]}
df = pd.DataFrame(data)

highlighted_df = highlight_above_average(df)
highlighted_df
